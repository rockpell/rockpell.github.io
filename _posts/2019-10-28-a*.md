int main() {
	int x = 0;
	int y = 0;
	int startX, startY;
	int endX, endY;
	int** data;
	string input;

	cin >> y >> x;
	// S는 출발지: 2
	// F는 도착지: 3
	// X는 장애물: 1
	// O는 빈칸: 0
	

	data = new int*[y];
	for (int i = 0; i < y; ++i)
		data[i] = new int[x];

	for (int i = 0; i < y; ++i) {
		cin >> input;
		for (int p = 0; p < x; ++p) {
			switch (input[p])
			{
			case 'O':
				data[i][p] = 0;
				break;
			case 'X':
				data[i][p] = 1;
				break;
			case 'S':
				data[i][p] = 2;
				startX = p;
				startY = i;
				break;
			case 'F':
				data[i][p] = 3;
				endX = p;
				endY = i;
				break;
			}
		}
	}

	Graph graph;

	graph.inputNodes(data, x, y);

	return 0;
}

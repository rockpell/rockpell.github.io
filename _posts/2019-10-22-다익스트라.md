# 다익스트라

참고사이트: [https://hsp1116.tistory.com/42]

### 배열을 이용한 다익스트라

```java
class Solution {
    public int solution(int N, int[][] road, int K) {
        int answer = 0;

        boolean[] visited = new boolean[N];
        int[] dist = new int[N];
        int[][] graph = new int[N][N];

        int u, v;

        for(int i = 0; i < dist.length; i++)
            dist[i] = Integer.MAX_VALUE;

        visited[0] = true; // 시작 정점 지정
        dist[0] = 0;

        for(int i = 0; i < road.length; i++){ // 입력값 저장
            u = road[i][0] - 1;
            v = road[i][1] - 1;
            if(graph[u][v] != 0){ // 같은 연결의 새로운 입력이 들어올시 거리가 더 짧은 것을 저장
                if(graph[u][v] > road[i][2]){
                    graph[u][v] = road[i][2];
                    graph[v][u] = road[i][2];
                }
            }
            else{
                graph[u][v] = road[i][2];
                graph[v][u] = road[i][2];
            }
        }

        for(int i = 0; i < N - 1; i++){
            int minValue = Integer.MAX_VALUE;
            int minIndex = 0;

            for(int p = 0; p < N; p++){ // 방문 안한 노드 중 최소값 찾기
                if(!visited[p]){
                    if(dist[p] < minValue){
                        minIndex = p;
                        minValue = dist[p];
                    }
                }
            }

            visited[minIndex] = true;

            for(int p = 0; p < N; p++){ // 인점 정점 중 가장 짧은 경로 찾기
                if(graph[minIndex][p] != 0){
                    if(dist[p] > dist[minIndex] + graph[minIndex][p])
                        dist[p] = dist[minIndex] + graph[minIndex][p];
                }
            }
        }

        for(int i = 0; i < N; i++){
            if(dist[i] <= K)
                ++answer;
        }

        return answer;
    }
}
```
